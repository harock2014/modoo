import csv

f=open('age.csv')
data=csv.reader(f)
result=[]
location = input('     ()   :')

for row in data:
    if location in row[0]:
        for i in row[3:]:
            result.append(int(i.replace(',','')))

import matplotlib.pyplot as plt

plt.style.use('ggplot')
plt.rc('font', family='Malgun Gothic')
plt.title(location+'  ')
plt.plot(result)
plt.show()

import csv
f=open('gender.csv')
data = csv.reader(f)

m = []
f = []

for row in data:
    if '' in row[0]:
        for i in range(0,101):
            m.append(int(row[i+3]))
            f.append(int(row[-(i+1)]))
            
f.reverse()

import matplotlib.pyplot as plt

plt.barh(range(101), m)
plt.barh(range(101), f)
plt.show()
